// vsechny URI dokumentu, ktery obsahujou text "Jan" (pokud je hotovy index podle system/config/db/jacob-test/collection.xconf)

declare default element namespace "http://www.tei-c.org/ns/1.0";
for $doc in collection("jacob-test")
let $qry := 
   <query>
     <bool>
       <term occur="should">Jan</term>
     </bool>
    </query>
where ft:query($doc//*, $qry)
return document-uri($doc)
